017package grails.plugins.quartz;
018
019import grails.plugins.quartz.config.TriggersConfigBuilder;
020import groovy.lang.Closure;
021import org.codehaus.groovy.grails.commons.AbstractInjectableGrailsClass;
022import org.codehaus.groovy.grails.commons.GrailsClassUtils;
023import org.quartz.JobExecutionContext;
024
025import java.util.HashMap;
026import java.util.Map;
027
028import static grails.plugins.quartz.GrailsJobClassConstants.*;
029
030
031/**
032 * Grails artifact class which represents a Quartz job.
033 *
034 * @author Micha?? K??ujszo
035 * @author Marcel Overdijk
036 * @author Sergey Nebolsin (nebolsin@gmail.com)
037 * @since 0.1
038 */
039public class DefaultGrailsJobClass extends AbstractInjectableGrailsClass implements GrailsJobClass {
040
041    public static final String JOB = "Job";
042    private Map triggers = new HashMap();
043
044
045    public DefaultGrailsJobClass(Class clazz) {
046        super(clazz, JOB);
047        evaluateTriggers();
048    }
049
050    private void evaluateTriggers() {
051        // registering additional triggersClosure from 'triggersClosure' closure if present
052        Closure triggersClosure = (Closure) GrailsClassUtils.getStaticPropertyValue(getClazz(), "triggers");
053
054        TriggersConfigBuilder builder = new TriggersConfigBuilder(getFullName());
055
056        if (triggersClosure != null) {
057            builder.build(triggersClosure);
058            triggers = (Map) builder.getTriggers();
059        }
060    }
061
062    public void execute() {
063        getMetaClass().invokeMethod(getReferenceInstance(), EXECUTE, new Object[]{});
064    }
065
066    public void execute(JobExecutionContext context) {
067        getMetaClass().invokeMethod(getReferenceInstance(), EXECUTE, new Object[]{context});
068    }
069
070    public String getGroup() {
071        String group = (String) getPropertyOrStaticPropertyOrFieldValue(GROUP, String.class);
072        if (group == null || "".equals(group)) return DEFAULT_GROUP;
073        return group;
074    }
075
076    public boolean isConcurrent() {
077        Boolean concurrent = getPropertyValue(CONCURRENT, Boolean.class);
078        return concurrent == null ? DEFAULT_CONCURRENT : concurrent;
079    }
080
081    public boolean isSessionRequired() {
082        Boolean sessionRequired = getPropertyValue(SESSION_REQUIRED, Boolean.class);
083        return sessionRequired == null ? DEFAULT_SESSION_REQUIRED : sessionRequired;
084    }
085
086    public boolean isDurability() {
087        Boolean durability = getPropertyValue(DURABILITY, Boolean.class);
088        return durability == null ? DEFAULT_DURABILITY : durability;
089    }
090
091    public boolean isRequestsRecovery() {
092        Boolean requestsRecovery = getPropertyValue(REQUESTS_RECOVERY, Boolean.class);
093        return requestsRecovery == null ? DEFAULT_REQUESTS_RECOVERY : requestsRecovery;
094    }
095
096    public String getDescription() {
097        String description = (String) getPropertyOrStaticPropertyOrFieldValue(DESCRIPTION, String.class);
098        if (description == null || "".equals(description)) return DEFAULT_DESCRIPTION;
099        return description;
100    }
101
102    public Map getTriggers() {
103        return triggers;
104    }
105}
