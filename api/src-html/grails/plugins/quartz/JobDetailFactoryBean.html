017package grails.plugins.quartz;
018
019import org.quartz.JobDetail;
020import org.springframework.beans.factory.FactoryBean;
021import org.springframework.beans.factory.InitializingBean;
022import org.springframework.beans.factory.annotation.Required;
023
024import static org.quartz.JobBuilder.newJob;
025
026/**
027 * Simplified version of Spring's <a href='http://static.springframework.org/spring/docs/2.5.x/api/org/springframework/scheduling/quartz/MethodInvokingJobDetailFactoryBean.html'>MethodInvokingJobDetailFactoryBean</a>
028 * that avoids issues with non-serializable classes (for JDBC storage).
029 *
030 * @author <a href='mailto:beckwithb@studentsonly.com'>Burt Beckwith</a>
031 * @author Sergey Nebolsin (nebolsin@gmail.com)
032 * @since 0.3.2
033 */
034public class JobDetailFactoryBean implements FactoryBean<JobDetail>, InitializingBean {
035    public static final transient String JOB_NAME_PARAMETER = "org.grails.plugins.quartz.grailsJobName";
036
037    // Properties
038    private GrailsJobClass jobClass;
039
040    // Returned object
041    private JobDetail jobDetail;
042
043    @Required
044    public void setJobClass(GrailsJobClass jobClass) {
045        this.jobClass = jobClass;
046    }
047
048    /**
049     * {@inheritDoc}
050     *
051     * @see org.springframework.beans.factory.InitializingBean#afterPropertiesSet()
052     */
053    public void afterPropertiesSet() {
054        String name = jobClass.getFullName();
055        if (name == null) {
056            throw new IllegalStateException("name is required");
057        }
058
059        String group = jobClass.getGroup();
060        if (group == null) {
061            throw new IllegalStateException("group is required");
062        }
063
064        // Consider the concurrent flag to choose between stateful and stateless job.
065        Class<? extends GrailsJobFactory.GrailsJob> clazz =
066                jobClass.isConcurrent() ? GrailsJobFactory.GrailsJob.class : GrailsJobFactory.StatefulGrailsJob.class;
067
068        // Build JobDetail instance.
069        jobDetail =
070                newJob(clazz)
071                        .withIdentity(name, group)
072                        .storeDurably(jobClass.isDurability())
073                        .requestRecovery(jobClass.isRequestsRecovery())
074                        .usingJobData(JOB_NAME_PARAMETER, name)
075                        .withDescription(jobClass.getDescription())
076                        .build();
077    }
078
079    /**
080     * {@inheritDoc}
081     *
082     * @see org.springframework.beans.factory.FactoryBean#getObject()
083     */
084    @Override
085    public JobDetail getObject() {
086        return jobDetail;
087    }
088
089    /**
090     * {@inheritDoc}
091     *
092     * @see org.springframework.beans.factory.FactoryBean#getObjectType()
093     */
094    @Override
095    public Class<JobDetail> getObjectType() {
096        return JobDetail.class;
097    }
098
099    /**
100     * {@inheritDoc}
101     *
102     * @see org.springframework.beans.factory.FactoryBean#isSingleton()
103     */
104    @Override
105    public boolean isSingleton() {
106        return true;
107    }
108}
