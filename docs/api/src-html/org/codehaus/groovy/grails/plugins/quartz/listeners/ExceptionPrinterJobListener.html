015    package org.codehaus.groovy.grails.plugins.quartz.listeners;
016    
017    import org.apache.commons.logging.Log;
018    import org.apache.commons.logging.LogFactory;
019    import org.quartz.JobExecutionContext;
020    import org.quartz.JobExecutionException;
021    import org.quartz.listeners.JobListenerSupport;
022    
023    /**
024     * JobListener implementation which logs an exceptions occured during job's execution.
025     *
026     * @author Sergey Nebolsin (nebolsin@gmail.com)
027     * 
028     * @since 0.2
029     */
030    public class ExceptionPrinterJobListener extends JobListenerSupport {
031            private static final transient Log LOG = LogFactory.getLog(ExceptionPrinterJobListener.class);
032    
033            public static final String NAME = "exceptionPrinterListener";
034    
035            public String getName() {
036                    return NAME;
037            }
038    
039            public void jobWasExecuted(JobExecutionContext context, JobExecutionException exception) {
040            if( exception != null )
041                LOG.error("Exception occured in job: " + context.getJobDetail().getFullName(), exception );
042        }
043    }
